function Bypass-USNJournal {
    param(
        [int]$FloodCount = 15000,          # Reduced to avoid filling the disk
        [int]$FileSize = 4096,
        [int]$Passes = 2,
        [string[]]$TargetDrives = $null,
        [switch]$DeleteTempFiles = $true,
        [switch]$ForceDeleteJournal = $false   # New: delete the journal entirely
    )

    Write-Host "    FloodCount = $FloodCount"
    Write-Host "    FileSize   = $FileSize bytes"
    Write-Host "    Passes     = $Passes"

    else { Write-Host "    TargetDrives = ALL fixed drives" }

    Write-Host ""

    # Delete Volume Shadow Copies (VSS)

    try { 
        $vss = Start-Process -FilePath "vssadmin.exe" -ArgumentList "delete shadows /all /quiet" -Wait -PassThru -NoNewWindow -ErrorAction SilentlyContinue
        if ($vss.ExitCode -eq 0) { Write-Host "    VSS deleted." -ForegroundColor Green } 
        else { Write-Host "    VSS deletion returned exit code $($vss.ExitCode) – may be normal." -ForegroundColor Yellow }
    } catch { Write-Host "    VSS deletion failed: $_" -ForegroundColor Yellow }
    Write-Host ""

    $drives = Get-WmiObject Win32_LogicalDisk -Filter "DriveType=3" -ErrorAction SilentlyContinue
    if (-not $drives) { $drives = Get-CimInstance Win32_LogicalDisk -Filter "DriveType=3" -ErrorAction SilentlyContinue }
    if (-not $drives) { Write-Host "[!] No fixed drives found. Exiting." -ForegroundColor Red; return }

        Write-Host "[*] Processing $driveLetter... " -ForegroundColor Cyan

        # --- Option 1: Delete journal entirely ---

            Write-Host "    [*] Deleting USN journal on $driveLetter`: ..." -ForegroundColor Yellow
            $del = Start-Process -FilePath "fsutil.exe" -ArgumentList "usn deletejournal /D $driveLetter`:" -Wait -PassThru -NoNewWindow -ErrorAction SilentlyContinue

                Write-Host "    [✓] Journal deleted. Recreating with default size..." -ForegroundColor Green

                    Write-Host "    [✓] Journal recreated." -ForegroundColor Green
                } else {
                    Write-Host "    [✗] Failed to recreate journal (exit $($create.ExitCode))." -ForegroundColor Red
                }
            } else {
                Write-Host "    [✗] Failed to delete journal (exit $($del.ExitCode))." -ForegroundColor Red
            }
            $anySuccess = $true
            continue   # Move to next drive
        }

        # --- Option 2: Shrink + Flood method ---
        for ($pass = 0; $pass -lt $Passes; $pass++) {
            Write-Host "    Pass $($pass+1)/$Passes..." -ForegroundColor Cyan

            # 1) Shrink journal to minimum (32KB) to force old records out
            Write-Host "        Shrinking journal..." -NoNewline
            $shrink = Start-Process -FilePath "fsutil.exe" -ArgumentList "usn createjournal m=32768 a=32768 $driveLetter`:" -Wait -PassThru -NoNewWindow -ErrorAction SilentlyContinue
            if ($shrink.ExitCode -ne 0) {
                Write-Host " FAILED (exit $($shrink.ExitCode))." -ForegroundColor Red
                # Try to get error output
                $err = Start-Process -FilePath "fsutil.exe" -ArgumentList "usn createjournal m=32768 a=32768 $driveLetter`:" -Wait -NoNewWindow -RedirectStandardError "temp_err.txt" -PassThru
                if (Test-Path "temp_err.txt") { Get-Content "temp_err.txt"; Remove-Item "temp_err.txt" -Force }
                continue
            }
            Write-Host " OK." -ForegroundColor Green

            # 2) Create flood of temp files
            Write-Host "        Creating $FloodCount files..." -NoNewline
            $created = 0
            $tempFiles = @()
            for ($i = 0; $i -lt $FloodCount; $i++) {
                $guid = [guid]::NewGuid().ToString()
                $tempName = "$driveLetter`:\JOURNAL_FLUSH_$guid.tmp"
                try {
                    [System.IO.File]::WriteAllBytes($tempName, (New-Object byte[] $FileSize))
                    $created++
                    $tempFiles += $tempName
                } catch {
                    # Stop if disk full or other error
                    break
                }
                # Progress every 1000 files to avoid slowing down
                if ($i % 1000 -eq 0 -and $i -gt 0) { Write-Host "." -NoNewline }
            }
            Write-Host " Done ($created created)." -ForegroundColor Green

            # 3) Shrink again to reset Next USN
            Write-Host "        Shrinking journal again..." -NoNewline
            $shrink2 = Start-Process -FilePath "fsutil.exe" -ArgumentList "usn createjournal m=32768 a=32768 $driveLetter`:" -Wait -PassThru -NoNewWindow -ErrorAction SilentlyContinue
            if ($shrink2.ExitCode -ne 0) {
                Write-Host " FAILED (exit $($shrink2.ExitCode))." -ForegroundColor Red
            } else {
                Write-Host " OK." -ForegroundColor Green
            }

            $anySuccess = $true
        }

        # --- Cleanup temp files (if requested) ---
        if ($DeleteTempFiles) {
            Write-Host "    Cleaning up temporary files on $driveLetter... " -NoNewline
            $pattern = "$driveLetter`:\JOURNAL_FLUSH_*.tmp"
            $files = Get-ChildItem -Path $pattern -File -ErrorAction SilentlyContinue
            if ($files) {
                $removed = 0
                foreach ($f in $files) {
                    try {
                        Remove-Item -Path $f.FullName -Force -ErrorAction Stop
                        $removed++
                    } catch {
                        Write-Host "Failed to delete $($f.Name)" -ForegroundColor Yellow
                    }
                }
                Write-Host "Deleted $removed files." -ForegroundColor Green
            } else {
                Write-Host "No files to delete." -ForegroundColor Gray
            }
        }

        Write-Host "[+] Finished processing $driveLetter" -ForegroundColor Green
        Write-Host ""
    }

    Write-Host ""
    if ($anySuccess) {
        Write-Host "[✓] USN Journal bypass completed!" -ForegroundColor Green
        if ($ForceDeleteJournal) {
            Write-Host "[*] The USN journal was deleted and recreated – all previous records are gone." -ForegroundColor Yellow
        } else {
            Write-Host "[*] Journal was shrunk and flooded – old records overwritten." -ForegroundColor Green
        }
        if ($DeleteTempFiles) {
            Write-Host "[*] Temporary files have been deleted." -ForegroundColor Green
        } else {
            Write-Host "[*] Temporary files remain at: X:\JOURNAL_FLUSH_*.tmp" -ForegroundColor Yellow
        }
    } else {
        Write-Host "[✗] USN Journal bypass failed on all targeted drives." -ForegroundColor Red
        Write-Host "    Common causes:" -ForegroundColor Yellow
        Write-Host "    - Not running as Administrator" -ForegroundColor Yellow
        Write-Host "    - Antivirus blocking file creation" -ForegroundColor Yellow
        Write-Host "    - Insufficient disk space for $FloodCount files" -ForegroundColor Yellow
    }
}
